# 破解root密码

开机，然后选择⬆️、⬇️方向键以取消倒计时，然后按照提示输入 e 键，进入编辑模式当中

找到“linux16”，按键 ↩︎ ，跳转到此段落的最后，输入一个空格键，再输入“rd.break”（有的系统是 single），之后按键 ctrl  + x or ctrl + s

至此进入修改密码的模式当中

mount -o remount,rw /sysroot

# TMOUT

设置系统的空闲等待时间 TMOUT.

有时，为了增强Linux系统的安全性，我们需要在用户输入空闲一段时间后自动断开，这个操作可以由设置 TMOUT 值来实现。

# 设置900秒内用户无操作就自动断开终端
echo "export TMOUT=900">> /etc/profile

# 将值设置为 readonly 防止用户更改
;notice:设置了readonly之后, 在当前shell下是无法取消的, 需要先将/etc/profile中readonly行注释掉或删除, logout后重新login.
echo "readonly TMOUT">> /etc/profile

# 读取配置文件
source /etc/profile

有时,设置了超时时间很短的话，当需要进行很多操作的时候，不希望经常超时断开，可以每次登录时候先在命令行下面执行下的命令，重启失效。

# 防止超时退出, 重启失效
TMOUT=0

"SELinux"
(default) SELinux禁止网络上对Samba服务器上的共享目录进行写操作

# method.01
# 临时禁用
setenforce 0
# 查看
getenforce

# method.02
# 设置SELinux放行标记了public_content_rw_t的内容
setsebool -P allow_smbd_anon_write=1
# 把欲共享的 <dir> 目录标记为public_content_rw_t
chcon -t public_content_rw_t <dir>
getsebool -a|grep samba
getsebool -a|grep smb
# 开启读写功能
setsebool samba_export_all_rw on
# 匿名共享
vim /etc/samba/smb.conf

[global]
				# 增加, 否则会要求认证身份
				map to guest = Bad User

[public]
				# 共享文件夹说明
				comment = Public Storage
				# 共享文件夹目录
				path = /opt/samba/public
				# 非只读
				read only = No
				# 可写
				writable = yes
				# 不公开
				public = No
				# 创建文件的权限
				create mask = 0777
				# 创建文件夹的权限
				directory mask = 0777
				# guest访问，无需密码
				guest ok = Yes
				# 可见
				browseable = Yes
				
esc
:wq

systemctl restart smb
# 设置文件夹权限, samba配置文件中的权限要与系统中共享文件夹的权限保持相同
chmod -R 777 /opt/samba/public

"SELinux"
setsebool -P allow_smbd_anon_write=1
chcon -t public_content_rw_t /opt/samba/public
setsebool samba_export_all_rw on
